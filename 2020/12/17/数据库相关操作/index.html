DDL - 数据定义语言

创建数据库

create database 数据库名 default charset utf8 (default 后面的语句，是修改字符集)

create database school default charset utf8;

删除数据库

drop 数据库名

drop database
--  if exists 存在才删除
drop database  if exists school;

切换数据库

use 数据库名

use school

创建表

create table 表名
(
字段名1 类型 约束,

字段名2 类型 约束,

…

字段名n 类型 约束,
primary key(主键名)

);

注：primary key 设置主键

-- 创建表
create table tb_student
(
stuid integer not null,
stuname varchar(10) not null,
stusex boolean default 1,
stubirth date,
stuaddr varchar(50),
primary key(stuid)
);

删除表

drop table 表名；

drop table school;

修改表

添加字段

alter table 表名 add column 字段名 类型 约束；

添加外键

alter table 表名add constraint 外键名
foreign key (外键) references 关联的外键表 (关联的外键名);

-- 修改学生表添加一个表示学生属于哪个学院的列
alter table tb_student add column colid integer not null;
-- 给students添加外键
-- 指定学生表中colid跟学院表的colid是什么关系
alter table tb_student add constraint  fk_student_colid 
foreign key (colid) references tb_college (colid);

DML - 数据操作语言

添加数据

  • 按位依次添加

  • 按字段名添加

  • 批量添加

    -- 按位依次添加
insert into tb_college values (1,'计算机学院','研究计算机科学与技术的学院');

insert into tb_college (colname,colintro) values ('计算机学院','研究计算机科学与技术的学院');

### 更新数据

update 表名 set 字段名1=修改的内容1，字段名2=修改的内容2 where 条件；

~~~ mysql
update tb_student set stusex=0 where stuid=1004;
update tb_student set stuaddr=null,stuname='李冰冰',stubirth='1980-3-25' where stuid=1004;

删除数据

delete from 表名 where 条件;

delete from tb_student where stuid=1003;

删除全表数据（截断表）
truncate table 表名 ;

truncate table tb_student;

DCL - 数据控制语言

创建用户

create user ‘用户名‘@’ip地址‘ identified by 密码;（%代表任意ip，localhost代表本机）

create user 'reddy'@'%'  identified by 'Reddyfan2020@';

删除用户

drop user ‘用户名‘@’ip地址‘

-- 删除用户
drop user 'reddy'@'%';

授权

  • 授予所有的权限 *.* 所有对象.所有权限

    grant all privileges *.* 'reddy'@'%' with grant option;

  • 授予用户指定数据库（对象）指定权限

    grant DELETE,UPDATE,SELECT on school.* to 'reddy'@'%';

召回权限

revoke all privileges on 数据名.* from ‘用户名‘@’ip地址‘

revoke all privileges on      school.* from  'reddy'@'%';
